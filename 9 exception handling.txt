public class ExceptionHandlingDemo {

    public static void main(String[] args) {

        // Scenario 1: Exception occurs and is caught
        try {
            System.out.println("Inside try block (Scenario 1)");
            int result = 10 / 0;
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Caught an ArithmeticException: " + e.getMessage());
        } finally {
            System.out.println("Finally block executed (Scenario 1)");
        }

        System.out.println("After try-catch-finally (Scenario 1)\n");

        // Scenario 2: No exception occurs
        try {
            System.out.println("Inside try block (Scenario 2)");
            int result = 10 / 2;
            System.out.println("Result: " + result + " (Scenario 2)");
        } catch (ArithmeticException e) {
            System.out.println("Caught an ArithmeticException: " + e.getMessage());
        } finally {
            System.out.println("Finally block executed (Scenario 2)");
        }

        System.out.println("After try-catch-finally (Scenario 2)\n");

        // Scenario 3: Exception occurs and is caught
        try {
            System.out.println("Inside try block (Scenario 3)");
            int[] numbers = {1, 2, 3};
            System.out.println(numbers[5]);
        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("Caught an ArrayIndexOutOfBoundsException: " + e.getMessage());
        } finally {
            System.out.println("Finally block executed (Scenario 3)");
        }

        System.out.println("After try-catch-finally (Scenario 3)");
    }
}
